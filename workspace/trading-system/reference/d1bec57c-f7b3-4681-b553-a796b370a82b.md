```python
def trading_decision(event_id, current_time):
    # 1. Time check
    hour = current_time.hour
    if hour in DANGER_HOURS:
        return False, "Danger hour"
    
    # 2. Window check (60% rule)
    event_duration = get_event_duration(event_id)
    elapsed = current_time - event_start_time
    if elapsed / event_duration < 0.60:
        return False, "Too early"
    
    # 3. Calculate real probability
    market_prob = get_polymarket_price(event_id)
    sentiment = get_sentiment_score(event_id)
    onchain = get_onchain_metric(event_id)
    
    real_prob = (
        market_prob * 0.40 +
        sentiment * 0.50 +
        onchain * 0.10
    )
    
    # 4. Check dissonance
    dissonance = abs(real_prob - market_prob)
    if dissonance < 0.30:
        return False, "Insufficient edge"
    
    # 5. Position sizing (Kelly)
    win_prob = get_historical_win_rate(hour)
    kelly = (win_prob * 3 - (1 - win_prob)) / 3
    
    if hour in WINNING_HOURS:
        position = portfolio_value * min(kelly, 0.15)
    elif hour in LEARNING_HOURS:
        position = portfolio_value * min(kelly, 0.05)
    else:
        return False, "Danger hour"
    
    return True, {
        'side': 'YES' if real_prob > market_prob else 'NO',
        'size': position,
        'confidence': win_prob,
        'dissonance': dissonance
    }
```
